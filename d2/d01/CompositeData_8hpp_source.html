92  template <class AspectT, typename... Args>
93  typename GetData<AspectT>::Type& create(Args&&... args)
94  {
95  using Data = typename GetData<AspectT>::Type;
96  using AspectType = typename GetAspect<AspectT>::Type;
97  using DataType = typename GetData<Aspect>::Type;
98 
99  std::unique_ptr<DataType>& data = this->mMap[typeid(AspectType)]
100  = make_unique<Data>(std::forward<Args>(args)...);
101  return static_cast<Data&>(*data);
102  }
103 
104  template <class AspectT>
105  typename GetData<AspectT>::Type* get()
106  {
107  using Data = typename GetData<AspectT>::Type;
108  using AspectType = typename GetAspect<AspectT>::Type;
109 
110  typename MapType::iterator it = this->mMap.find(typeid(AspectType));
111  if(this->mMap.end() == it)
112  return nullptr;
113 
114  return static_cast<Data*>(it->second.get());
115  }
116 
117  template <class AspectT>
118  const typename GetData<AspectT>::Type* get() const
119  {
120  return const_cast<CompositeData<MapType, GetData>*>(this)->get<AspectT>();
121  }
122 
123  template <class AspectT, typename... Args>
124  typename GetData<AspectT>::Type& getOrCreate(Args&&... args)
125  {
126  using Data = typename GetData<AspectT>::Type;
127  using AspectType = typename GetAspect<AspectT>::Type;
128 
129  auto& it = this->mMap.insert(
130  std::make_pair<std::type_index, std::unique_ptr<Data>>(
131  typeid(AspectType), nullptr));
132 
133  const bool exists = !it.second;
134  if(!exists)
135  it.first = make_unique<Data>(std::forward<Args>(args)...);
136 
137  return static_cast<Data&>(*it.first);
138  }
139 
140  template <class AspectT>
141  bool has() const
142  {
143  return (get<AspectT>() != nullptr);
144  }
145 };
