239  TemplateBodyNodePtr<BodyNodeT> lock() const
240  {
241  if(nullptr == mLocker)
242  return nullptr;
243 
244  // We do not use the expired() function here, because we want to ensure that
245  // the Skeleton's reference count remains up while we create the strong
246  // BodyNodePtr that we're going to return.
247  std::lock_guard<std::mutex> lock(mLocker->mMutex);
248  std::shared_ptr<const Skeleton> skeleton = mLocker->mSkeleton.lock();
249  if(nullptr == skeleton)
250  return nullptr;
251 
252  return TemplateBodyNodePtr<BodyNodeT>(mPtr);
253  }
